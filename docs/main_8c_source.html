125  }
126 
127  /* Read status register to acknowledge interrupt
128  * (can be disabled by changing LINK/LOOP mode in ADXL_REG_ACT_INACT_CTL)
129  * TODO this can perhaps fix the bug where too much movement breaks interrupt wake-up ...
130  * also change in ADXL_readValues! */
131  if (ADXL_getTriggered())
132  {
133 
134 #if DEBUGGING == 1 /* DEBUGGING */
135  dbprintln_color("INT-PD7 triggered!", 4);
136 #endif /* DEBUGGING */
137 
138  ADXL_enableSPI(true); /* Enable SPI functionality */
139  ADXL_ackInterrupt(); /* Acknowledge ADXL interrupt */
140  ADXL_enableSPI(false); /* Disable SPI functionality */

151  /* Set the index to put the measurements in */
152  data.index = 0;
153 
154  /* TODO: Remove this variable later? */
155  bool notBroken = false;
156 
157  while(1)

159  switch(MCUstate)
160  {
161  case INIT:
162  {
163  CHIP_Init(); /* Initialize chip */
164 
165 #if DEBUGGING == 1 /* DEBUGGING */

167  dbprint_INIT(DBG_UART, DBG_UART_LOC, false, false);
168 #else /* Regular Happy Gecko pinout */
169  dbprint_INIT(USART1, 4, true, false); /* VCOM */
170  //dbprint_INIT(USART1, 0, false, false); /* US1_TX = PC0 */
171 #endif /* Board pinout selection */
172 #endif /* DEBUGGING */
173 
174  USTIMER_Init(); /* Initialize timer (necessary for DS18B20 logic) */
175 
176  led(true); /* Enable (and initialize) LED */
177 
178  initGPIOwakeup(); /* Initialize GPIO wake-up */
179 
180  initADC(BATTERY_VOLTAGE); /* Initialize ADC to read battery voltage */
181 
182  /* Disable RN2483 */
183  GPIO_PinModeSet(PM_RN2483_PORT, PM_RN2483_PIN, gpioModePushPull, 0);
184 
185  /* Initialize accelerometer */
186  if (true)
187  {
188  /* Initialize the accelerometer */
189  initADXL();
190 
191  /* Set the measurement range */
192  ADXL_configRange(ADXL_RANGE_4G);
193 
194  /* Configure ODR */
195  ADXL_configODR(ADXL_ODR_12_5);
196 
197  /* Read and display values forever */
198  //ADXL_readValues();
199 
200  /* Configure activity detection on INT1 */
201  ADXL_configActivity(3); /* [g] */
202 
203  /* Enable measurements */
204  ADXL_enableMeasure(true);
205 
206  delay(100); /* TODO: Weird INT behavour, try to fix this with link-looped mode! */
207 
208  /* ADXL gives interrupt, capture this */
209  ADXL_ackInterrupt(); /* Acknowledge ADXL interrupt */
210 
211  /* Disable SPI after the initializations */
212  ADXL_enableSPI(false);
213 
214  /* ADXL give interrupt, capture this */
215  ADXL_setTriggered(false); /* Reset variable again */
216  }
217 
218  MCUstate = MEASURE;
219  } break;
220 
221  case MEASURE:
222  {
223  // TODO: add check if max data is filled here?
224 
225  led(true); /* Enable LED */
226 
227  delay(500);
228 
229  checkInterrupts();
230 
231  /* Measure and store the external temperature (a measurement takes about 550 ms) */
232  data.extTemp[data.index] = readTempDS18B20();
233 
234  /* Measure and store the battery voltage */
235  data.voltage[data.index] = readADC(BATTERY_VOLTAGE);
236 
237  /* Measure and store the internal temperature */
238  data.intTemp[data.index] = readADC(INTERNAL_TEMPERATURE);
239 
240 #if DEBUGGING == 1 /* DEBUGGING */
241  dbinfoInt("Temperature: ", data.extTemp[data.index]*1000, "");
242  dbinfoInt("Battery voltage: ", data.voltage[data.index], "");
243  dbinfoInt("Internal temperature: ", data.intTemp[data.index], "");
244 #endif /* DEBUGGING */
245 
246  notBroken = checkCable(); /* Check the cable */
247 
248 #if DEBUGGING == 1 /* DEBUGGING */
249  if (notBroken) dbinfo("Cable still intact");
250  else dbcrit("Cable broken!");
251 #endif /* DEBUGGING */
252 
253  /* Increase the index to put the next measurements in */
254  data.index++;
255 
256  led(false); /* Disable LED */
257 
258  MCUstate = SLEEP;
259  } break;
260 
261  case SEND:
262  {
263 
264 #if DEBUGGING == 1 /* DEBUGGING */
265  dbwarn("Normally we send the data now.");
266 #endif /* DEBUGGING */
267 
268  // TODO: Call data sending method here
269 
270  /* Reset the index to put the measurements in */
271  data.index = 0;
272 
273  MCUstate = MEASURE;
274  } break;
275 
276  case SLEEP:
277  {
278  GPIO_PinModeSet(DBG_RXD_PORT, DBG_RXD_PIN, gpioModeDisabled, 0);
279 
280  sleep(WAKE_UP_PERIOD_S); /* Go to sleep for xx seconds */
281 
282  GPIO_PinModeSet(DBG_RXD_PORT, DBG_RXD_PIN, gpioModeInput, 0);
283 
284  MCUstate = WAKEUP;
285  } break;
286 
287  case WAKEUP:
288  {
289  /* Decide if 6 measurements are taken or not and react accordingly */
290  if (data.index == 6) MCUstate = SEND;
291  else MCUstate = MEASURE;
292  } break;
293 
294  default:
295  {
296  error(0);
297  } break;
298  }
299  }
