353 /**************************************************************************//**
354  * @brief
355  * Configure the accelerometer to work in activity threshold mode.
356  *
357  * @details
358  * Route activity detector to INT1 pin using INTMAP1, isolate bits
359  * and write settings to both threshold registers.
360  *
361  * @param[in] gThreshold
362  * Threshold [g].
363  *****************************************************************************/
364 void ADXL_configActivity (uint8_t gThreshold)
365 {
366  /* Map activity detector to INT1 pin */
367  writeADXL(ADXL_REG_INTMAP1, 0b00010000); /* Bit 4 selects activity detector */
368 
369  /* Enable referenced activity threshold mode (last two bits) */
370  writeADXL(ADXL_REG_ACT_INACT_CTL, 0b00000011);
371 
372  /* Convert g value to "codes":
373  * THRESH_ACT [codes] = Threshold Value [g] × Scale Factor [LSB per g] */
374  uint16_t threshold;
375 
376  if (range == 0) threshold = gThreshold * 1000;
377  else if (range == 1) threshold = gThreshold * 500;
378  else if (range == 2) threshold = gThreshold * 250;
379  else
380  {
381 
382 #ifdef DEBUGGING /* DEBUGGING */
383  dbcrit("Range wrong, can't set gThreshold!");
384 #endif /* DEBUGGING */
385 
386  error(5);
387  }
388 
389  /* Isolate bits using masks and shifting */
390  uint8_t low = (threshold & 0b0011111111);
391  uint8_t high = (threshold & 0b1100000000) >> 8;
392 
393  /* Set threshold register values (total: 11bit unsigned) */
394  writeADXL(ADXL_REG_THRESH_ACT_L, low); /* 7:0 bits used */
395  writeADXL(ADXL_REG_THRESH_ACT_H, high); /* 2:0 bits used */
396 
397 #ifdef DEBUGGING /* DEBUGGING */
398  dbinfoInt("ADXL362: Activity configured: ", gThreshold, " g");
399 #endif /* DEBUGGING */
400 
401 }

462 /**************************************************************************//**
463  * @brief
464  * Initialize USART0 in SPI mode according to the settings required
465  * by the accelerometer.
466  *
467  * @details
468  * Configure pins, configure USART0 in SPI mode, route
469  * the pins, enable USART0 and set CS high.
470  * Necessary clocks are enabled in a previous method.
471  *
472  * @note
473  * This is a static method because it's only internally used in this file
474  * and called by other methods if necessary.
475  *****************************************************************************/
476 static void initADXL_SPI (void)
477 {
478  /* Configure GPIO */
479  /* In the case of gpioModePushPull", the last argument directly sets the pin state */
480  GPIO_PinModeSet(ADXL_CLK_PORT, ADXL_CLK_PIN, gpioModePushPull, 0); /* US0_CLK is push pull */
481  GPIO_PinModeSet(ADXL_NCS_PORT, ADXL_NCS_PIN, gpioModePushPull, 1); /* US0_CS is push pull */
482  GPIO_PinModeSet(ADXL_MOSI_PORT, ADXL_MOSI_PIN, gpioModePushPull, 1); /* US0_TX (MOSI) is push pull */
483  GPIO_PinModeSet(ADXL_MISO_PORT, ADXL_MISO_PIN, gpioModeInput, 1); /* US0_RX (MISO) is input */
484 
485  /* Start with default config */
486  USART_InitSync_TypeDef config = USART_INITSYNC_DEFAULT;
487 
488  /* Modify some settings */
489  config.enable = false; /* making sure to keep USART disabled until we've set everything up */
490  config.refFreq = 0; /* USART/UART reference clock assumed when configuring baud rate setup. Set to 0 to use the currently configured reference clock. */
491  config.baudrate = 4000000; /* CLK freq is 1 MHz (1000000) */
492  config.databits = usartDatabits8; /* master mode */
493  config.master = true; /* master mode */
494  config.msbf = true; /* send MSB first */
495  config.clockMode = usartClockMode0; /* clock idle low, sample on rising/first edge (Clock polarity/phase mode = CPOL/CPHA) */
496  config.prsRxEnable = false; /* If enabled: Enable USART Rx via PRS. */
497  config.autoTx = false; /* If enabled: Enable AUTOTX mode. Transmits as long as RX is not full. Generates underflows if TX is empty. */
498  config.autoCsEnable = false; /* CS pin controlled by hardware, not firmware */
499 
500  /* Initialize USART0/1 with the configured parameters */
501  USART_InitSync(ADXL_SPI, &config);
502 
503  /* Set the pin location for USART0/1 (before: USART_ROUTE_LOCATION_LOC0) */
504  ADXL_SPI->ROUTE = USART_ROUTE_CLKPEN | USART_ROUTE_CSPEN | USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | ADXL_SPI_LOC;
505 
506  /* Enable USART0/1 */
507  USART_Enable(ADXL_SPI, usartEnable);
508 
509  /* Set CS high (active low!) */
510  GPIO_PinOutSet(gpioPortE, 13);
511 }

514 /**************************************************************************//**
515  * @brief
516  * Soft reset accelerometer handler.
517  *
518  * @details
519  * If the first ID check fails, the MCU is put on hold for one second
520  * and the ID gets checked again.
521  *
522  * @note
523  * This is a static method because it's only internally used in this file
524  * and called by other methods if necessary.
525  *****************************************************************************/
526 static void resetHandlerADXL (void)
527 {
528  uint8_t retries = 0;
529 
530  /* Soft reset ADXL */
531  softResetADXL();
532 
533  /* First try to get the correct ID failed */
534  if (!checkID_ADXL())
535  {
536  retries++;
537 
538  delay(1000);
539 
540  /* Soft reset */
541  softResetADXL();
542 
543  /* Second try to get the correct ID failed */
544  if (!checkID_ADXL())
545  {
546  retries++;
547 
548  delay(1000);
549 
550  /* Soft reset */
551  softResetADXL();
552 
553  /* Third try to get the correct ID failed
554  * resorting to "hard" reset! */
555  if (!checkID_ADXL())
556  {
557  retries++;
558 
559  powerADXL(false);
560 
561  delay(1000);
562 
563  powerADXL(true);
564 
565  delay(1000);
566 
567  /* Soft reset */
568  softResetADXL();
569 
570  /* Last try to get the correct ID failed */
571  if (!checkID_ADXL())
572  {
573  error(1);
574  }
575  }
576  }
577  }
578 
579 #ifdef DEBUGGING /* DEBUGGING */
580  if (retries < 2) dbinfoInt("ADXL362 initialized (", retries, " soft reset retries)");
581  else dbwarnInt("ADXL362 initialized (had to \"hard reset\", ", retries, " soft reset retries)");
582 #endif /* DEBUGGING */
583 
584 }

588  * @brief
589  * Read an SPI byte from the accelerometer (8 bits) using a given address.
590  *
591  * @note
592  * This is a static method because it's only internally used in this file
593  * and called by other methods if necessary.
594  *
595  * @param[in] address
596  * The register address to read from.
597  *
598  * @return
599  * The response (one byte, uint8_t).
600  *****************************************************************************/
601 static uint8_t readADXL (uint8_t address)
602 {
603  uint8_t response;
604 
605  /* Set CS low (active low!) */
606  GPIO_PinOutClear(ADXL_NCS_PORT, ADXL_NCS_PIN);
607 
608  /* 3-byte operation according to datasheet */
609  USART_SpiTransfer(ADXL_SPI, 0x0B); /* "read" instruction */
610  USART_SpiTransfer(ADXL_SPI, address); /* Address */
611  response = USART_SpiTransfer(ADXL_SPI, 0x00); /* Read response */
612 
613  /* Set CS high */
614  GPIO_PinOutSet(ADXL_NCS_PORT, ADXL_NCS_PIN);
615 
616  return (response);
617 }

668  USART_SpiTransfer(ADXL_SPI, ADXL_REG_XDATA); /* Address */

670  XYZDATA[1] = USART_SpiTransfer(ADXL_SPI, 0x00); /* Read response */
671  XYZDATA[2] = USART_SpiTransfer(ADXL_SPI, 0x00); /* Read response */
672

674  GPIO_PinOutSet(ADXL_NCS_PORT, ADXL_NCS_PIN);

758 static int32_t convertGRangeToGValue (int8_t sensorValue)

766  else if (range == 1) return ((2*4*1000 / 255) * sensorValue);
767  else if (range == 2) return ((2*8*1000 / 255) * sensorValue);
768  else
769  {

772  dbcrit("Range wrong, can't calculate mg value!");
773 #endif /* DEBUGGING */
