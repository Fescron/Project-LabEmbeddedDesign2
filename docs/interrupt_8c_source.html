51 /**************************************************************************//**
52  * @brief
53  * Initialize GPIO wake-up functionality.
54  *
55  * @details
56  * Initialize buttons PB0 and PB1 on falling-edge interrupts and
57  * ADXL_INT1 on rising-edge interrupts.
58  *****************************************************************************/
59 void initGPIOwakeup (void)
60 {
61  /* Enable necessary clocks (just in case) */
62  CMU_ClockEnable(cmuClock_HFPER, true); /* GPIO is a High Frequency Peripheral */
63  CMU_ClockEnable(cmuClock_GPIO, true);
64 
65  /* Configure PB0 and PB1 as input with glitch filter enabled, last argument sets pull direction */
66  GPIO_PinModeSet(PB0_PORT, PB0_PIN, gpioModeInputPullFilter, 1);
67  GPIO_PinModeSet(PB1_PORT, PB1_PIN, gpioModeInputPullFilter, 1);
68 
69  /* Configure ADXL_INT1 as input, the last argument enables the filter */
70  GPIO_PinModeSet(ADXL_INT1_PORT, ADXL_INT1_PIN, gpioModeInput, 1);
71 
72 
73  /* Clear all odd pin interrupt flags (just in case)
74  * NVIC_ClearPendingIRQ(GPIO_ODD_IRQn); would also work but is less "readable" */
75  GPIO_IntClear(0xAAAA);
76 
77  /* Clear all even pin interrupt flags (just in case)
78  * NVIC_ClearPendingIRQ(GPIO_EVEN_IRQn); would also work but is less "readable" */
79  GPIO_IntClear(0x5555);
80 
81  /* All pending interrupts can be cleared with GPIO_IntClear(0xFFFF); */
82 
83  /* Enable IRQ for even numbered GPIO pins */
84  NVIC_EnableIRQ(GPIO_EVEN_IRQn);
85 
86  /* Enable IRQ for odd numbered GPIO pins */
87  NVIC_EnableIRQ(GPIO_ODD_IRQn);
88 
89  /* Enable falling-edge interrupts for PB pins */
90  GPIO_ExtIntConfig(PB0_PORT, PB0_PIN, PB0_PIN, false, true, true);
91  GPIO_ExtIntConfig(PB1_PORT, PB1_PIN, PB1_PIN, false, true, true);
92 
93  /* Enable rising-edge interrupts for ADXL_INT1 */
94  GPIO_ExtIntConfig(ADXL_INT1_PORT, ADXL_INT1_PIN, ADXL_INT1_PIN, true, false, true);
95 
96 #ifdef DEBUGGING /* DEBUGGING */
97  dbinfo("GPIO wake-up initialized");
98 #endif /* DEBUGGING */
99 
100 }

165  *****************************************************************************/
166 void GPIO_EVEN_IRQHandler(void)
167 {
168  /* Disable the counter */
169  RTC_Enable(false);
170 
171  /* Read interrupt flags */
172  uint32_t flags = GPIO_IntGet();
173 
174  /* Check if PB1 is pushed */
175  if (flags == 0x400) PB1_triggered = true;
176  else
177  {
178 
179 #ifdef DEBUGGING /* DEBUGGING */
180  dbcrit("Unknown even-numbered IRQ pin triggered!");
181 #endif /* DEBUGGING */
182 
183  error(2);
184  }
185 
186  /* Clear all even pin interrupt flags */
187  GPIO_IntClear(0x5555);
188 }
