97 /**************************************************************************//**
98  * @brief
99  * Method to read the battery voltage or internal temperature.
100  *
101  * @param[in] temperature
102  * @li `false` - Read the VDD voltage.
103  * @li `true` - Read the internal temperature.
104  *
105  * @return
106  * The measured battery voltage or internal temperature.
107  *****************************************************************************/
108 uint32_t readADC (bool temperature)
109 {
110  uint32_t value = 0;
111 
112  /* Enable necessary clock */
113  CMU_ClockEnable(cmuClock_ADC0, true);
114 
115  /* Change ADC settings if necessary */
116  if (temperature)
117  {
118  if (initSingle.input != adcSingleInpTemp)
119  {
120  initSingle.input = adcSingleInpTemp; /* Internal temperature */
121  ADC_InitSingle(ADC0, &initSingle);
122 
123 #if DEBUGGING == 1 /* DEBUGGING */
124  dbinfo("ADC0 re-initialized for internal temperature");
125 #endif /* DEBUGGING */
126 
127  }
128  }
129  else
130  {
131  if (initSingle.input != adcSingleInpVDDDiv3)
132  {
133  initSingle.input = adcSingleInpVDDDiv3; /* Internal VDD/3 */
134  ADC_InitSingle(ADC0, &initSingle);
135 
136 #if DEBUGGING == 1 /* DEBUGGING */
137  dbinfo("ADC0 re-initialized for VBAT");
138 #endif /* DEBUGGING */
139 
140  }
141  }
142 
143  /* Set variable false just in case */
144  adcConversionComplete = false;
145 
146  /* Start single ADC conversion */
147  ADC_Start(ADC0, adcStartSingle);
148 
149  /* Wait in EM1 until the conversion is completed */
150  while (!adcConversionComplete) EMU_EnterEM1();
151 
152  /* Get the ADC value */
153  value = ADC_DataSingleGet(ADC0);
154 
155  /* Disable used clock */
156  CMU_ClockEnable(cmuClock_ADC0, false);
157 
158  /* Calculate final value according to parameter */
159  if (temperature)
160  {
161  float32_t ft = convertToCelsius(value)*1000;
162  value = (uint32_t) ft;
163  }
164  else
165  {
166  float32_t fv = value * 3.75 / 4.096;
167  value = (uint32_t) fv;
168  }
169 
170  return (value);
171 }

174  /**************************************************************************//**
175  * @brief
176  * Method to convert an ADC value to a temperature value.
177  *
178  * @note
179  * This is a static method because it's only internally used in this file
180  * and called by other methods if necessary.
181  *
182  * @param[in] adcSample
183  * The ADC sample to convert to a temperature value.
184  *
185  * @return
186  * The converted temperature value.
187  *****************************************************************************/
188 static float32_t convertToCelsius (int32_t adcSample)
189 {
190  float32_t temp;
191 
192  /* Factory calibration temperature from device information page. */
193  int32_t cal_temp_0 = ((DEVINFO->CAL & _DEVINFO_CAL_TEMP_MASK)
194  >> _DEVINFO_CAL_TEMP_SHIFT);
195 
196  /* Factory calibration value from device information page. */
197  int32_t cal_value_0 = ((DEVINFO->ADC0CAL2 & _DEVINFO_ADC0CAL2_TEMP1V25_MASK)
198  >> _DEVINFO_ADC0CAL2_TEMP1V25_SHIFT);
199 
200  /* Temperature gradient (from datasheet) */
201  float32_t t_grad = -6.27;
202 
203  temp = (cal_temp_0 - ((cal_value_0 - adcSample) / t_grad));
204 
205  return (temp);
206 }
