95 /**************************************************************************//**
96  * @brief
97  * Add a battery voltage measurement to the LPP packet, disguised as an
98  * *Analog Input* packet (2 bytes). The channel is defined by `LPP_VBAT_CHANNEL`
99  * and is `0x10`.
100  *
101  * @param[in] b
102  * The pointer to the LPP pointer.
103  *
104  * @param[in] vbat
105  * The measured battery voltage.
106  *
107  * @return
108  * @li `true` - Successfully added the data to the LoRaWAN packet.
109  * @li `false` - Couldn't add the data to the LoRaWAN packet.
110  *****************************************************************************/
111 bool LPP_AddVBAT (LPP_Buffer_t *b, int16_t vbat)
112 {
113  uint8_t space = b->length - b->fill;
114  if (space < LPP_ANALOG_INPUT_SIZE) return (false);
115 
116  b->buffer[b->fill++] = LPP_VBAT_CHANNEL;
117  b->buffer[b->fill++] = LPP_ANALOG_INPUT;
118  b->buffer[b->fill++] = (uint8_t)((0xFF00 & vbat) >> 8);
119  b->buffer[b->fill++] = (uint8_t)(0x00FF & vbat);
120 
121  return (true);
122 }
