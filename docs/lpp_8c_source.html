101 /**************************************************************************//**
102  * @brief
103  * Add measurement data to the LPP packet following the *custom message
104  * convention* to save bytes to send.
105  *
106  * @details
107  * This is what each added byte represents, **in the case of `2` measurements**:
108  * - **byte 0:** Amount of measurements
109  * - **byte 1:** Battery voltage channel (`LPP_VBAT_CHANNEL = 0x10`)
110  * - **byte 2:** LPP analog input type (`LPP_ANALOG_INPUT = 0x02`)
111  * - **byte 3-4:** A battery voltage measurement
112  * - **byte 5-6:** Another battery voltage measurement (in the case of `2` measurements)
113  * - ...
114  * - **byte 7:** Internal temperature channel (`LPP_TEMPERATURE_CHANNEL_INT = 0x11`)
115  * - **byte 8:** LPP temperature type (`LPP_TEMPERATURE = 0x67`)
116  * - **byte 9-10:** An internal temperature measurement
117  * - **byte 11-12:** Another internal temperature measurement (in the case of `2` measurements)
118  * - ...
119  * - **byte 13:** External temperature channel (`LPP_TEMPERATURE_CHANNEL_EXT = 0x12`)
120  * - **byte 14:** LPP temperature type (`LPP_TEMPERATURE = 0x67`)
121  * - **byte 15-16:** An external temperature measurement
122  * - **byte 17-18:** Another external temperature measurement (in the case of `2` measurements)
123  * - ...
124  *
125  * If we have **6 measurements** we need **43 bytes**:
126  * - `1 byte` to hold the amount of measurements
127  * - `2 bytes` to hold the battery voltage channel and LPP analog input type
128  * - `6*2bytes` to hold the battery voltage measurements
129  * - `2 bytes` to hold the internal temperature channel and LPP temperature type
130  * - `6*2bytes` to hold the internal temperature measurements
131  * - `2 bytes` to hold the external temperature channel and LPP temperature type
132  * - `6*2bytes` to hold the external temperature measurements
133  *
134  * @param[in] b
135  * The pointer to the LPP pointer.
136  *
137  * @param[in] data
138  * The struct which contains the measurements to send using LoRaWAN.
139  *
140  * @return
141  * @li `true` - Successfully added the data to the LoRaWAN packet.
142  * @li `false` - Couldn't add the data to the LoRaWAN packet.
143  *****************************************************************************/
144 bool LPP_AddMeasurements (LPP_Buffer_t *b, MeasurementData_t data)
145 {
146  /* Calculate free space in the buffer */
147  uint8_t space = b->length - b->fill;
148 
149  /* Calculate necessary space, first byte holds the amount of measurements */
150  uint8_t necessarySpace = 1;
151 
152  /* Add space necessary for the battery voltage and internal and external temperature measurements.
153  * (1 byte for the channel ID, 1 byte for the data type, 2 bytes for each measurement) */
154  necessarySpace += 3*(2+(2*(data.index)));
155 
156  /* Return `false` if we don't have the necessary space available */
157  if (space < necessarySpace) return (false);
158 
159  /* Fill the first byte with the amount of measurements */
160  b->buffer[b->fill++] = data.index;
161 
162  /* Fill the next bytes with battery voltage measurements */
163  b->buffer[b->fill++] = LPP_VBAT_CHANNEL;
164  b->buffer[b->fill++] = LPP_ANALOG_INPUT;
165 
166  for (uint8_t i = 0; i < data.index; i++)
167  {
168  /* Convert battery voltage value (should represent 0.01 signed ) */
169  int16_t batteryLPP = (int16_t)(round((float)data.voltage[i]/10));
170 
171  b->buffer[b->fill++] = (uint8_t)((0xFF00 & batteryLPP) >> 8);
172  b->buffer[b->fill++] = (uint8_t)(0x00FF & batteryLPP);
173  }
174 
175  /* Fill the next bytes with internal temperature measurements */
176  b->buffer[b->fill++] = LPP_TEMPERATURE_CHANNEL_INT;
177  b->buffer[b->fill++] = LPP_TEMPERATURE;
178 
179  for (uint8_t i = 0; i < data.index; i++)
180  {
181  /* Convert temperature sensor value (should represent 0.1 °C Signed MSB) */
182  int16_t intTempLPP = (int16_t)(round((float)data.intTemp[i]/100));
183 
184  b->buffer[b->fill++] = (uint8_t)((0xFF00 & intTempLPP) >> 8);
185  b->buffer[b->fill++] = (uint8_t)(0x00FF & intTempLPP);
186  }
187 
188  /* Fill the next bytes with external temperature measurements */
189  b->buffer[b->fill++] = LPP_TEMPERATURE_CHANNEL_EXT;
190  b->buffer[b->fill++] = LPP_TEMPERATURE;
191 
192  for (uint8_t i = 0; i < data.index; i++)
193  {
194  /* Convert temperature sensor value (should represent 0.1 °C Signed MSB) */
195  int16_t extTempLPP = (int16_t)(round((float)data.extTemp[i]/100));
196 
197  b->buffer[b->fill++] = (uint8_t)((0xFF00 & extTempLPP) >> 8);
198  b->buffer[b->fill++] = (uint8_t)(0x00FF & extTempLPP);
199  }
200 
201  return (true);
202 }

204 /**************************************************************************//**
205  * @brief
206  * Add a value to indicate that a storm has been detected to the LPP packet
207  * following the *custom message convention*.
208  *
209  * @details
210  * This is what each added byte represents:
211  * - **byte 0:** Amount of measurements (in this case always one)
212  * - **byte 1:** *Storm detected* channel (`LPP_STORM_CHANNEL = 0x13`)
213  * - **byte 2:** LPP digital input type (`LPP_DIGITAL_INPUT = 0x00`)
214  * - **byte 3:** The `stormDetected` value
215  *
216  * **We always need 4 bytes.**
217  *
218  * @param[in] b
219  * The pointer to the LPP pointer.
220  *
221  * @param[in] stormDetected
222  * If a storm is detected using the accelerometer or not.
223  *
224  * @return
225  * @li `true` - Successfully added the data to the LoRaWAN packet.
226  * @li `false` - Couldn't add the data to the LoRaWAN packet.
227  *****************************************************************************/
228 bool LPP_AddStormDetected (LPP_Buffer_t *b, uint8_t stormDetected)
229 {
230  /* Calculate free space in the buffer */
231  uint8_t space = b->length - b->fill;
232 
233  /* Return `false` if we don't have the necessary space available */
234  if (space < LPP_DIGITAL_INPUT_SIZE + 1) return (false); /* "+1": One extra byte for the amount of measurements */
235 
236  /* Fill the first byte with the amount of measurements (in this case always one) */
237  b->buffer[b->fill++] = 0x01;
238 
239  /* Fill the next bytes following the default LPP packet convention */
240  b->buffer[b->fill++] = LPP_STORM_CHANNEL;
241  b->buffer[b->fill++] = LPP_DIGITAL_INPUT;
242  b->buffer[b->fill++] = stormDetected;
243 
244  return (true);
245 }
