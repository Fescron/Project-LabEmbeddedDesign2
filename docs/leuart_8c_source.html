107 void setupDma(void){
108  /* DMA configuration structs */
109  DMA_Init_TypeDef dmaInit;
110  DMA_CfgChannel_TypeDef rxChnlCfg;
111  DMA_CfgChannel_TypeDef txChnlCfg;
112  DMA_CfgDescr_TypeDef rxDescrCfg;
113  DMA_CfgDescr_TypeDef txDescrCfg;
114 
115  /* Initializing the DMA */
116  dmaInit.hprot = 0;
117  dmaInit.controlBlock = dmaControlBlock;
118  DMA_Init(&dmaInit);
119 
120  /* RX DMA setup*/
121  /* Set the interrupt callback routine */
122  dmaCallBack[DMA_CHANNEL_RX].cbFunc = basicRxComplete;
123  /* Callback doesn't need userpointer */
124  dmaCallBack[DMA_CHANNEL_RX].userPtr = NULL;
125 
126  /* Setting up channel for TX*/
127  rxChnlCfg.highPri = false; /* Can't use with peripherals */
128  rxChnlCfg.enableInt = true; /* Enabling interrupt to refresh DMA cycle*/
129  /*Setting up DMA transfer trigger request*/
130  rxChnlCfg.select = DMAREQ_LEUART0_RXDATAV; //DMAREQ_LEUART0_RXDATAV;
131  /* Setting up callback function to refresh descriptors*/
132  rxChnlCfg.cb = &(dmaCallBack[DMA_CHANNEL_RX]);
133  DMA_CfgChannel(DMA_CHANNEL_RX, &rxChnlCfg);
134 
135  /* Setting up channel descriptor */
136  /* Destination is buffer, increment ourselves */
137  rxDescrCfg.dstInc = dmaDataIncNone;
138  /* Source is LEUART_RX register and transfers 8 bits each time */
139  rxDescrCfg.srcInc = dmaDataIncNone;
140  rxDescrCfg.size = dmaDataSize1;
141  /* Default setting of DMA arbitration*/
142  rxDescrCfg.arbRate = dmaArbitrate1;
143  rxDescrCfg.hprot = 0;
144  /* Configure primary descriptor */
145  DMA_CfgDescr(DMA_CHANNEL_RX, true, &rxDescrCfg);
146  DMA_CfgDescr(DMA_CHANNEL_RX, false, &rxDescrCfg);
147 
148  /* TX DMA setup*/
149  /* Set the interrupt callback routine */
150  dmaCallBack[DMA_CHANNEL_TX].cbFunc = basicTxComplete;
151  /* Callback doesn't need userpointer */
152  dmaCallBack[DMA_CHANNEL_TX].userPtr = NULL;
153 
154  /* Setting up channel for TX*/
155  txChnlCfg.highPri = false; /* Can't use with peripherals */
156  txChnlCfg.enableInt = true; /* Enabling interrupt to refresh DMA cycle*/
157  /*Setting up DMA transfer trigger request*/
158  txChnlCfg.select = DMAREQ_LEUART0_TXBL; //DMAREQ_LEUART0_RXDATAV;
159  /* Setting up callback function to refresh descriptors*/
160  txChnlCfg.cb = &(dmaCallBack[DMA_CHANNEL_TX]);
161  DMA_CfgChannel(DMA_CHANNEL_TX, &txChnlCfg);
162 
163  /* Setting up channel descriptor */
164  /* Destination is LEUART_Tx register and doesn't move */
165  txDescrCfg.dstInc = dmaDataIncNone;
166  /* Source is LEUART_TX register and transfers 8 bits each time */
167  txDescrCfg.srcInc = dmaDataInc1;
168  txDescrCfg.size = dmaDataSize1;
169  /* Default setting of DMA arbitration*/
170  txDescrCfg.arbRate = dmaArbitrate1;
171  txDescrCfg.hprot = 0;
172  /* Configure primary descriptor */
173  DMA_CfgDescr(DMA_CHANNEL_TX, true, &txDescrCfg);
174 }

190 static void setupLeuart(void){
191  /* Allocate timer for Timeout detection*/
192  /* RTCDRV_AllocateTimer(&xTimerForTimeout); */
193 
194  /* Enable peripheral clocks */
195  CMU_ClockEnable(cmuClock_HFPER, true);
196  /* Configure GPIO pins */
197  CMU_ClockEnable(cmuClock_GPIO, true);
198  /* To avoid false start, configure output as high */
199  GPIO_PinModeSet(RN2483_TX_PORT, RN2483_TX_PIN, gpioModePushPull, 1);
200  GPIO_PinModeSet(RN2483_RX_PORT, RN2483_RX_PIN, gpioModeInput, 0);
201 
202  LEUART_Init_TypeDef init = LEUART_INIT_DEFAULT; // Default config is fine
203  init.baudrate = 4800;
204 
205  /* Enable CORE LE clock in order to access LE modules */
206  CMU_ClockEnable(cmuClock_CORELE, true);
207 
208  /* Select LFRCO for LEUARTs (and wait for it to stabilize) */
209  CMU_ClockSelectSet(cmuClock_LFB, cmuSelect_LFXO);
210  CMU_ClockEnable(cmuClock_LEUART0, true);
211 
212  /* Do not prescale clock */
213  CMU_ClockDivSet(cmuClock_LEUART0, cmuClkDiv_1);
214 
215  /* Configure LEUART */
216  init.enable = leuartDisable;
217 
218  LEUART_Init(RN2483_UART, &init);
219 
220  /* Enable pins at default location */
221  RN2483_UART->ROUTE = LEUART_ROUTE_RXPEN | LEUART_ROUTE_TXPEN | RN2483_UART_LOC;
222 
223  /* Set RXDMAWU to wake up the DMA controller in EM2 */
224  LEUART_RxDmaInEM2Enable(RN2483_UART, true);
225 
226  /* Clear previous RX interrupts */
227  LEUART_IntClear(RN2483_UART, LEUART_IF_RXDATAV);
228  NVIC_ClearPendingIRQ(LEUART0_IRQn);
229 
230  /* Finally enable it */
231  LEUART_Enable(RN2483_UART, leuartEnable);
232 }
