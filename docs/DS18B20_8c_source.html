98 /**************************************************************************//**
99  * @brief
100  * Get a temperature value from the DS18B20.
101  *
102  * @details
103  * USTimer gets initialized, the sensor gets powered, the data-transmission
104  * takes place, the timer gets de-initialized to disable the clocks and interrupts,
105  * the data and power pin get disabled and finally the read values are converted
106  * to an `int32_t` value.@n
107  * **Negative temperatures work fine.**
108  *
109  * @return
110  * The read temperature data.
111  *****************************************************************************/
112 int32_t readTempDS18B20 (void)
113 {
114  /* Timeout counter */
115  uint16_t counter = 0;
116 
117  /* Variable to indicate if a conversion has been completed */
118  bool conversionCompleted = false;
119 
120  /* Variable to hold raw data bytes */
121  uint8_t rawDataFromDS18B20Arr[9] = {0x0, 0x0, 0x0, 0x0, 0x0, 0x0, 0x0, 0x0, 0x0};
122 
123  /* Initialize timer
124  * Initializing and disabling the timer again adds about 40 µs active time but should conserve sleep energy... */
125  USTIMER_Init();
126 
127  /* Initialize and power VDD pin */
128  powerDS18B20(true);
129 
130  /* Power-up delay of 5 ms */
131  delay(5);
132 
133  /* Initialize communication and only continue if successful */
134  if (init_DS18B20())
135  {
136  writeByteToDS18B20(0xCC); /* 0xCC = "Skip Rom" (address all devices on the bus simultaneously without sending out any ROM code information) */
137  writeByteToDS18B20(0x44); /* 0x44 = "Convert T" */
138 
139  /* MASTER now generates "read time slots", the DS18B20 will write HIGH to the bus if the conversion is completed
140  * The datasheet gives the following directions for time slots, but reading bytes also seems to work...
141  * - Read time slots have a 60 µs duration and 1 µs recovery between slots
142  * - After the master pulls the line low for 1 µs, the data is valid for up to 15 µs */
143  while ((counter < TIMEOUT_CONVERSION) && !conversionCompleted)
144  {
145  uint8_t testByte = readByteFromDS18B20();
146  if (testByte > 0) conversionCompleted = true;
147 
148  counter++;
149  }
150 
151  /* Exit the function if the maximum waiting time was reached */
152  if (counter == TIMEOUT_CONVERSION)
153  {
154 
155 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
156  dbcrit("Waiting time for DS18B20 conversion reached!");
157 #endif /* DEBUG_DBPRINT */
158 
159  /* Disable interrupts and turn off the clock to the underlying hardware timer. */
160  USTIMER_DeInit();
161 
162  /* Disable data pin (otherwise we got a "sleep" current of about 330 µA due to the on-board 10k pull-up) */
163  GPIO_PinModeSet(TEMP_DATA_PORT, TEMP_DATA_PIN, gpioModeDisabled, 0);
164 
165  /* Disable the VDD pin */
166  powerDS18B20(false);
167 
168  error(29);
169 
170  /* Exit function */
171  return (0);
172 
173  }
174 #if DBPRINT_TIMEOUT == 1 /* DBPRINT_TIMEOUT */
175  else
176  {
177 
178 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
179  dbwarnInt("DS18B20 conversion (", counter, ")");
180 #endif /* DEBUG_DBPRINT */
181 
182  }
183 #endif /* DBPRINT_TIMEOUT */
184 
185  init_DS18B20(); /* Initialize communication */
186  writeByteToDS18B20(0xCC); /* 0xCC = "Skip Rom" */
187  writeByteToDS18B20(0xBE); /* 0xCC = "Read Scratchpad" */
188 
189  /* Read the bytes */
190  for (uint8_t i = 0; i < 9; i++) rawDataFromDS18B20Arr[i] = readByteFromDS18B20();
191 
192  /* Disable interrupts and turn off the clock to the underlying hardware timer. */
193  USTIMER_DeInit();
194 
195  /* Disable data pin (otherwise we got a "sleep" current of about 330 µA due to the on-board 10k pull-up) */
196  GPIO_PinModeSet(TEMP_DATA_PORT, TEMP_DATA_PIN, gpioModeDisabled, 0);
197 
198  /* Disable the VDD pin */
199  powerDS18B20(false);
200 
201  /* Return the converted byte */
202  return (convertTempData(rawDataFromDS18B20Arr[0], rawDataFromDS18B20Arr[1]));
203  }
204  else
205  {
206  /* Disable interrupts and turn off the clock to the underlying hardware timer. */
207  USTIMER_DeInit();
208 
209  /* Disable data pin (otherwise we got a "sleep" current of about 330 µA due to the on-board 10k pull-up) */
210  GPIO_PinModeSet(TEMP_DATA_PORT, TEMP_DATA_PIN, gpioModeDisabled, 0);
211 
212  /* Disable the VDD pin */
213  powerDS18B20(false);
214 
215  /* Exit function */
216  return (0);
217  }
218 }

221 /**************************************************************************//**
222  * @brief
223  * Enable or disable the power to the temperature sensor.
224  *
225  * @details
226  * This method also initializes the pin-mode if necessary.
227  *
228  * @note
229  * This is a static method because it's only internally used in this file
230  * and called by other methods if necessary.
231  *
232  * @param[in] enabled
233  * @li `true` - Enable the GPIO pin connected to the VDD pin of the temperature sensor.
234  * @li `talse` - Disable the GPIO pin connected to the VDD pin of the temperature sensor.
235  *****************************************************************************/
236 static void powerDS18B20 (bool enabled)
237 {
238  /* Enable necessary clocks (just in case) */
239  CMU_ClockEnable(cmuClock_HFPER, true); /* GPIO is a High Frequency Peripheral */
240  CMU_ClockEnable(cmuClock_GPIO, true);
241 
242  /* Initialize VDD pin if not already the case */
243  if (!DS18B20_VDD_initialized)
244  {
245  /* In the case of gpioModePushPull", the last argument directly sets the pin state */
246  GPIO_PinModeSet(TEMP_VDD_PORT, TEMP_VDD_PIN, gpioModePushPull, enabled);
247 
248  DS18B20_VDD_initialized = true;
249  }
250  else
251  {
252  if (enabled) GPIO_PinOutSet(TEMP_VDD_PORT, TEMP_VDD_PIN); /* Enable VDD pin */
253  else GPIO_PinOutClear(TEMP_VDD_PORT, TEMP_VDD_PIN); /* Disable VDD pin */
254  }
255 }

258 /**************************************************************************//**
259  * @brief
260  * Initialize communication to the DS18B20.
261  *
262  * @note
263  * This is a static method because it's only internally used in this file
264  * and called by other methods if necessary.
265  *
266  * @return
267  * @li `true` - *Presence* pulse detected in time.
268  * @li `false` - No *presence* pulse detected.
269  *****************************************************************************/
270 static bool init_DS18B20 (void)
271 {
272  /* Timeout counter */
273  uint32_t counter = 0;
274 
275  /* MASTER RESET: Pull data line LOW for at least 480 µs (Master TX) */
276  GPIO_PinModeSet(TEMP_DATA_PORT, TEMP_DATA_PIN, gpioModePushPull, 0); /* gpioModePushPull: Last argument directly sets the pin state */
277  USTIMER_DelayIntSafe(480);
278 
279  /* Change pin-mode to input - External pull-up resistor pulls data line back HIGH */
280  GPIO_PinModeSet(TEMP_DATA_PORT, TEMP_DATA_PIN, gpioModeInput, 0);
281 
282  /* Check if the line becomes LOW (~ wait while it stays high) during the maximum waiting time
283  * The DS18B20 should detect the data line rising due to the pull-up resistor, waits 15 - 50 µs
284  * and then pulls the line back LOW (for 60 - 240 µs) to indicate it's PRESENCE */
285  while ((counter < TIMEOUT_INIT) && (GPIO_PinInGet(TEMP_DATA_PORT, TEMP_DATA_PIN) == 1)) counter++;
286 
287  /* Exit the function if the maximum waiting time was reached */
288  if (counter == TIMEOUT_INIT)
289  {
290 
291 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
292  dbcrit("No DS18B20 presence pulse detected in time!");
293 #endif /* DEBUG_DBPRINT */
294 
295  error(28);
296 
297  return (false);
298  }
299 #if DBPRINT_TIMEOUT == 1 /* DBPRINT_TIMEOUT */
300  else
301  {
302 
303 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
304  dbwarnInt("DS18B20 INIT (", counter, ")");
305 #endif /* DEBUG_DBPRINT */
306 
307  }
308 #endif /* DBPRINT_TIMEOUT */
309 
310  /* Master RX should be at least 480 µs */
311  USTIMER_DelayIntSafe(480);
312 
313  return (true);
314 }

317 /**************************************************************************//**
318  * @brief
319  * Write a byte (`uint8_t`) to the DS18B20.
320  *
321  * @note
322  * This is a static method because it's only internally used in this file
323  * and called by other methods if necessary.
324  *
325  * @param[in] data
326  * The data to write to the DS18B20.
327  *****************************************************************************/
328 static void writeByteToDS18B20 (uint8_t data)
329 {
330  /* In the case of gpioModePushPull", the last argument directly sets the pin state */
331  GPIO_PinModeSet(TEMP_DATA_PORT, TEMP_DATA_PIN, gpioModePushPull, 0);
332 
333  /* Write the byte, bit by bit */
334  for (uint8_t i = 0; i < 8; i++)
335  {
336  /* Check if we need to write a "1" */
337  if (data & 0x01)
338  {
339  GPIO_PinOutClear(TEMP_DATA_PORT, TEMP_DATA_PIN);
340 
341  /* 5 µs delay should be called here but this loop works fine too... */
342  for (uint8_t i=0; i<5; i++);
343 
344  GPIO_PinOutSet(TEMP_DATA_PORT, TEMP_DATA_PIN);
345  USTIMER_DelayIntSafe(60);
346  }
347  /* If not, write a "0" */
348  else
349  {
350  GPIO_PinOutClear(TEMP_DATA_PORT, TEMP_DATA_PIN);
351  USTIMER_DelayIntSafe(60);
352  GPIO_PinOutSet(TEMP_DATA_PORT, TEMP_DATA_PIN);
353 
354  /* 5 µs delay should be called here but this loop works fine too... */
355  for (uint8_t i=0; i<5; i++);
356  }
357  /* Right shift bits once */
358  data >>= 1;
359  }
360 
361  /* Set data line high */
362  GPIO_PinOutSet(TEMP_DATA_PORT, TEMP_DATA_PIN);
363 }

366 /**************************************************************************//**
367  * @brief
368  * Read a byte (`uint8_t`) from the DS18B20.
369  *
370  * @note
371  * This is a static method because it's only internally used in this file
372  * and called by other methods if necessary.
373  *
374  * @return
375  * The byte read from the DS18B20.
376  *****************************************************************************/
377 static uint8_t readByteFromDS18B20 (void)
378 {
379  /* Data to eventually return */
380  uint8_t data = 0x0;
381 
382  /* Read the byte, bit by bit */
383  for (uint8_t i = 0; i < 8; i++)
384  {
385  /* Change pin-mode to input */
386  GPIO_PinModeSet(TEMP_DATA_PORT, TEMP_DATA_PIN, gpioModeInput, 0);
387 
388  /* Right shift bits once */
389  data >>= 1;
390 
391  /* If the line is high, OR the first bit of the data:
392  * 0x80 = 1000 0000 */
393  if (GPIO_PinInGet(TEMP_DATA_PORT, TEMP_DATA_PIN)) data |= 0x80;
394 
395  /* In the case of gpioModePushPull", the last argument directly sets the pin state */
396  GPIO_PinModeSet(TEMP_DATA_PORT, TEMP_DATA_PIN, gpioModePushPull, 1);
397 
398  /* Wait some time before going into next loop */
399  USTIMER_DelayIntSafe(70);
400  }
401  return (data);
402 }

405 /**************************************************************************//**
406  * @brief
407  * Convert temperature data.
408  *
409  * @note
410  * This is a static method because it's only internally used in this file
411  * and called by other methods if necessary.
412  *
413  * @param[in] tempLS
414  * Least significant byte.
415  *
416  * @param[in] tempMS
417  * Most significant byte.
418  *
419  * @return
420  * The converted temperature data.
421  *****************************************************************************/
422 static int32_t convertTempData (uint8_t tempLS, uint8_t tempMS)
423 {
424  uint16_t rawDataMerge;
425  uint16_t reverseRawDataMerge;
426 
427  int32_t finalTemperature;
428 
429  /* Check if it is a negative temperature value
430  * 0xF8 = 0b1111 1000 */
431  if (tempMS & 0xF8)
432  {
433  rawDataMerge = tempMS;
434 
435  /* Left shift 8 times */
436  rawDataMerge <<= 8;
437 
438  /* Add the second part */
439  rawDataMerge += tempLS;
440 
441  /* Invert the value since we have a negative temperature */
442  reverseRawDataMerge = ~rawDataMerge;
443 
444  /* Calculate the final temperature */
445  finalTemperature = -(reverseRawDataMerge + 1) * 62.5;
446  }
447  /* We're dealing with a positive temperature */
448  else
449  {
450  rawDataMerge = tempMS;
451 
452  /* Left shift 8 times */
453  rawDataMerge <<= 8;
454 
455  /* Add the second part */
456  rawDataMerge += tempLS;
457 
458  /* Calculate the final temperature */
459  finalTemperature = rawDataMerge * 62.5;
460  }
461 
462  return (finalTemperature);
463 }
