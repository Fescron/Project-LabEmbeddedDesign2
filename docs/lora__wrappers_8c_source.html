110 /**************************************************************************//**
111  * @brief
112  * Send measured battery voltages and internal and external temperatures
113  * to the cloud using LoRaWAN.
114  *
115  * @details
116  * The code detects if not all six measurement fields are filled in the array
117  * and reacts accordingly using the `index` field.
118  *
119  * @param[in] data
120  * The struct which contains the measurements to send using LoRaWAN.
121  *
122  * @param[in] stormDetected
123  * A value to indicate if a storm is detected using the accelerometer.
124  *****************************************************************************/
125 void sendMeasurements (MeasurementData_t data, bool stormDetected)
126 {
127  /* Initialize and fill LPP-formatted payload
128  * 6*VBAT (4) + 6*intTemp (4) + 6*extTemp (4) + 1*stormDetected (3) = 75 bytes */
129  if (!LPP_InitBuffer(&appData, 75)) error(21);
130 
131  /* Add battery voltage measurements to LPP payload */
132  for (uint8_t i = 0; i <= data.index-1; i++)
133  {
134  int16_t batteryLPP = (int16_t)(round((float)data.voltage[i]/10));
135  if (!LPP_AddVBAT(&appData, batteryLPP)) error(22);
136  }
137 
138  /* Add internal temperature measurements to LPP payload */
139  for (uint8_t i = 0; i <= data.index-1; i++)
140  {
141  /* Convert temperature sensor value (should represent 0.1 °C Signed MSB) */
142  int16_t intTempLPP = (int16_t)(round((float)data.intTemp[i]/100));
143  if (!LPP_AddIntTemp(&appData, intTempLPP)) error(23);
144  }
145 
146  /* Add external temperature measurements to LPP payload */
147  for (uint8_t i = 0; i <= data.index-1; i++)
148  {
149  /* Convert temperature sensor value (should represent 0.1 °C Signed MSB) */
150  int16_t extTempLPP = (int16_t)(round((float)data.extTemp[i]/100));
151  if (!LPP_AddExtTemp(&appData, extTempLPP)) error(24);
152  }
153 
154  /* Add "storm" value */
155  if (!LPP_AddStormDetected(&appData, stormDetected)) error(25);
156 
157  /* Send LPP-formatted payload */
158  LoRaStatus_t test = LoRa_SendLppBuffer(appData, LORA_UNCONFIMED); /* TODO: fix this, can't send (buffer too large?) */
159  if (LoRa_SendLppBuffer(appData, LORA_UNCONFIMED) != SUCCESS) error(26);
160 }
