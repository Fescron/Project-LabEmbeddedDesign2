112 /**************************************************************************//**
113  * @brief
114  * Send measured battery voltages and internal and external temperatures
115  * to the cloud using LoRaWAN.
116  *
117  * @details
118  * The measurements get added to the LPP packet following the *custom message
119  * convention* to save bytes to send.
120  *
121  * @param[in] data
122  * The struct which contains the measurements to send using LoRaWAN.
123  *****************************************************************************/
124 void sendMeasurements (MeasurementData_t data)
125 {
126  /* Initialize LPP-formatted payload
127  * For 6 measurements we need a max amount of 43 bytes (see `LPP_AddMeasurements` method documentation for the calculation) */
128  if (!LPP_InitBuffer(&appData, 43)) error(31);
129 
130  /* Add measurements to the LPP packet using the custom convention to save bytes send */
131  if (!LPP_AddMeasurements(&appData, data)) error(32);
132 
133  /* Send custom LPP-like-formatted payload */
134  if (LoRa_SendLppBuffer(appData, LORA_UNCONFIMED) != SUCCESS) error(33);
135 }

206 /**************************************************************************//**
207  * @brief
208  * Send ONE measured battery voltage, internal and external temperature,
209  * `stormDetected`, `cableBroken` and `status` value to the cloud using
210  * LoRaWAN. This method uses the deprecated methods to test if the data
211  * gets send correctly.
212  *
213  * @param[in] data
214  * The struct which contains the measurements to send using LoRaWAN.
215  *****************************************************************************/
216 void sendTest (MeasurementData_t data)
217 {
218  /* Initialize LPP-formatted payload - We need 21 bytes */
219  if (!LPP_InitBuffer(&appData, 21)) error(43);
220 
221  /* Add measurements to the LPP packet */
222  int16_t batteryLPP = (int16_t)(round((float)data.voltage[0]/10));
223  if (!LPP_deprecated_AddVBAT(&appData, batteryLPP)) error(44);
224 
225  int16_t intTempLPP = (int16_t)(round((float)data.intTemp[0]/100));
226  if (!LPP_deprecated_AddIntTemp(&appData, intTempLPP)) error(45);
227 
228  int16_t extTempLPP = (int16_t)(round((float)data.extTemp[0]/100));
229  if (!LPP_deprecated_AddExtTemp(&appData, extTempLPP)) error(46);
230 
231  if (!LPP_deprecated_AddStormDetected(&appData, true)) error(47);
232  if (!LPP_deprecated_AddCableBroken(&appData, true)) error(48);
233  if (!LPP_deprecated_AddStatus(&appData, 11)) error(49);
234 
235  /* Send LPP-formatted payload */
236  if (LoRa_SendLppBuffer(appData, LORA_UNCONFIMED) != SUCCESS) error(50);
