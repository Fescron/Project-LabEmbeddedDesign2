128 /**************************************************************************//**
129  * @brief
130  * Initialize USARTx.
131  *
132  * @param[in] pointer
133  * Pointer to USARTx.
134  *
135  * @param[in] location
136  * Location for pin routing.
137  *
138  * @param[in] vcom
139  * @li `true` - Isolation switch enabled by setting `PA9` high so the **Virtual COM port (CDC)** can be used.
140  * @li `false` - Isolation switch disabled on the Happy Gecko board.
141  *
142  * @param[in] interrupts
143  * @li `true` - Enable interrupt functionality.
144  * @li `false` - No interrupt functionality is initialized.
145  *****************************************************************************/
146 void dbprint_INIT (USART_TypeDef* pointer, uint8_t location, bool vcom, bool interrupts)
147 {
148  /* Store the pointer in the global variable */
149  dbpointer = pointer;
150 
151  /*
152  * USART_INITASYNC_DEFAULT:
153  * config.enable = usartEnable // Specifies whether TX and/or RX is enabled when initialization is completed
154  * // (Enable RX/TX when initialization is complete).
155  * config.refFreq = 0 // USART/UART reference clock assumed when configuring baud rate setup
156  * // (0 = Use current configured reference clock for configuring baud rate).
157  * config.baudrate = 115200 // Desired baudrate (115200 bits/s).
158  * config.oversampling = usartOVS16 // Oversampling used (16x oversampling).
159  * config.databits = usartDatabits8 // Number of data bits in frame (8 data bits).
160  * config.parity = usartNoParity // Parity mode to use (no parity).
161  * config.stopbits = usartStopbits1 // Number of stop bits to use (1 stop bit).
162  * config.mvdis = false // Majority Vote Disable for 16x, 8x and 6x oversampling modes (Do not disable majority vote).
163  * config.prsRxEnable = false // Enable USART Rx via PRS (Not USART PRS input mode).
164  * config.prsRxCh = 0 // Select PRS channel for USART Rx. (Only valid if prsRxEnable is true - PRS channel 0).
165  * config.autoCsEnable = false // Auto CS enabling (Auto CS functionality enable/disable switch - disabled).
166  */
167 
168  USART_InitAsync_TypeDef config = USART_INITASYNC_DEFAULT;
169 
170  /* Enable oscillator to GPIO*/
171  CMU_ClockEnable(cmuClock_GPIO, true);
172 
173 
174  /* Enable oscillator to USARTx modules */
175  if (dbpointer == USART0)
176  {
177  CMU_ClockEnable(cmuClock_USART0, true);
178  }
179  else if (dbpointer == USART1)
180  {
181  CMU_ClockEnable(cmuClock_USART1, true);
182  }
183 
184 
185  /* Set PA9 (EFM_BC_EN) high if necessary to enable the isolation switch */
186  if (vcom)
187  {
188  GPIO_PinModeSet(gpioPortA, 9, gpioModePushPull, 1);
189  GPIO_PinOutSet(gpioPortA, 9);
190  }
191 
192 
193  /* Set pin modes for UART TX and RX pins */
194  if (dbpointer == USART0)
195  {
196  switch (location)
197  {
198  case 0:
199  GPIO_PinModeSet(gpioPortE, 11, gpioModeInput, 0); /* RX */
200  GPIO_PinModeSet(gpioPortE, 10, gpioModePushPull, 1); /* TX */
201  break;
202  case 2:
203  GPIO_PinModeSet(gpioPortC, 10, gpioModeInput, 0); /* RX */
204  /* No TX pin in this mode */
205  break;
206  case 3:
207  GPIO_PinModeSet(gpioPortE, 12, gpioModeInput, 0); /* RX */
208  GPIO_PinModeSet(gpioPortE, 13, gpioModePushPull, 1); /* TX */
209  break;
210  case 4:
211  GPIO_PinModeSet(gpioPortB, 8, gpioModeInput, 0); /* RX */
212  GPIO_PinModeSet(gpioPortB, 7, gpioModePushPull, 1); /* TX */
213  break;
214  case 5:
215  case 6:
216  GPIO_PinModeSet(gpioPortC, 1, gpioModeInput, 0); /* RX */
217  GPIO_PinModeSet(gpioPortC, 0, gpioModePushPull, 1); /* TX */
218  break;
219  /* default: */
220  /* No default */
221  }
222  }
223  else if (dbpointer == USART1)
224  {
225  switch (location)
226  {
227  case 0:
228  GPIO_PinModeSet(gpioPortC, 1, gpioModeInput, 0); /* RX */
229  GPIO_PinModeSet(gpioPortC, 0, gpioModePushPull, 1); /* TX */
230  break;
231  case 2:
232  case 3:
233  GPIO_PinModeSet(gpioPortD, 6, gpioModeInput, 0); /* RX */
234  GPIO_PinModeSet(gpioPortD, 7, gpioModePushPull, 1); /* TX */
235  break;
236  case 4:
237  GPIO_PinModeSet(gpioPortA, 0, gpioModeInput, 0); /* RX */
238  GPIO_PinModeSet(gpioPortF, 2, gpioModePushPull, 1); /* TX */
239  break;
240  case 5:
241  GPIO_PinModeSet(gpioPortC, 2, gpioModeInput, 0); /* RX */
242  GPIO_PinModeSet(gpioPortC, 1, gpioModePushPull, 1); /* TX */
243  break;
244  /* default: */
245  /* No default */
246  }
247  }
248 
249 
250  /* Initialize USART asynchronous mode */
251  USART_InitAsync(dbpointer, &config);
252 
253  /* Route pins */
254  switch (location)
255  {
256  case 0:
257  dbpointer->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC0;
258  break;
259  case 1:
260  dbpointer->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC1;
261  break;
262  case 2:
263  dbpointer->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC2;
264  break;
265  case 3:
266  dbpointer->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC3;
267  break;
268  case 4:
269  dbpointer->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC4;
270  break;
271  case 5:
272  dbpointer->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC5;
273  break;
274  case 6:
275  dbpointer->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_LOC6;
276  break;
277  default:
278  dbpointer->ROUTE |= USART_ROUTE_TXPEN | USART_ROUTE_RXPEN | USART_ROUTE_LOCATION_DEFAULT;
279  }
280 
281  /* Enable interrupts if necessary and print welcome string (and make an alert sound in the console) */
282  if (interrupts)
283  {
284  /* Initialize USART interrupts */
285 
286  /* RX Data Valid Interrupt Enable
287  * Set when data is available in the receive buffer. Cleared when the receive buffer is empty. */
288  USART_IntEnable(dbpointer, USART_IEN_RXDATAV);
289 
290  /* TX Complete Interrupt Enable
291  * Set when a transmission has completed and no more data is available in the transmit buffer.
292  * Cleared when a new transmission starts. */
293  USART_IntEnable(dbpointer, USART_IEN_TXC);
294 
295  if (dbpointer == USART0)
296  {
297  /* Enable USART interrupts */
298  NVIC_EnableIRQ(USART0_RX_IRQn);
299  NVIC_EnableIRQ(USART0_TX_IRQn);
300  }
301  else if (dbpointer == USART1)
302  {
303  /* Enable USART interrupts */
304  NVIC_EnableIRQ(USART1_RX_IRQn);
305  NVIC_EnableIRQ(USART1_TX_IRQn);
306  }
307 
308  /* Print welcome string */
309  dbprint(COLOR_RESET);
310  dbprintln("\a\r\f### UART initialized (interrupt mode) ###");
311  dbinfo("This is an info message.");
312  dbwarn("This is a warning message.");
313  dbcrit("This is a critical error message.");
314  dbprintln("### Start executing programmed code ###\n");
315 
316  /* Set TX Complete Interrupt Flag (transmission has completed and no more data
317  * is available in the transmit buffer) */
318  USART_IntSet(dbpointer, USART_IFS_TXC);
319  }
320  /* Print welcome string (and make an alert sound in the console) if not in interrupt mode */
321  else
322  {
323  dbprint(COLOR_RESET);
324  dbprintln("\a\r\f### UART initialized (no interrupts) ###");
325  dbinfo("This is an info message.");
326  dbwarn("This is a warning message.");
327  dbcrit("This is a critical error message.");
328  dbprintln("### Start executing programmed code ###\n");
329  }
330 }

360 /**************************************************************************//**
361  * @brief
362  * Print a string (char array) to USARTx.
363  *
364  * @note
365  * If the input is not a string (ex.: `"Hello world!"`) but a char array,
366  * the input message (array) needs to end with NULL (`"\0"`)!
367  *
368  * @param[in] message
369  * The string to print to USARTx.
370  *****************************************************************************/
371 void dbprint (char *message)
372 {
373  /* "message[i] != 0" makes "uint32_t length = strlen(message)"
374  * not necessary (given string MUST be terminated by NULL for this to work) */
375  for (uint32_t i = 0; message[i] != 0; i++)
376  {
377  USART_Tx(dbpointer, message[i]);
378  }
379 }

1040 /**************************************************************************//**
1041  * @brief
1042  * Convert a `uint32_t` value to a hexadecimal char array (string).
1043  *
1044  * @note
1045  * This is a static method because it's only internally used in this file
1046  * and called by other methods if necessary.
1047  *
1048  * @param[out] buf
1049  * The buffer to put the resulting string in.@n
1050  * **This needs to have a length of 9: `char buf[9];`!**
1051  *
1052  * @param[in] value
1053  * The `uint32_t` value to convert to a string.
1054  *
1055  * @param[in] spacing
1056  * @li `true` - Add spacing between the eight HEX chars to make two groups of four.
1057  * @li `false` - Don't add spacing between the eight HEX chars.
1058  *****************************************************************************/
1059 static void uint32_to_charHex (char *buf, uint32_t value, bool spacing)
1060 {
1061  /* 4 nibble HEX representation */
1062  if (value <= 0xFFFF)
1063  {
1064  /* Only get necessary nibble by ANDing with a mask and
1065  * shifting one nibble (4 bits) per position */
1066  buf[0] = TO_HEX(((value & 0xF000) >> 12));
1067  buf[1] = TO_HEX(((value & 0x0F00) >> 8 ));
1068  buf[2] = TO_HEX(((value & 0x00F0) >> 4 ));
1069  buf[3] = TO_HEX( (value & 0x000F) );
1070  buf[4] = '\0'; /* NULL termination character */
1071  }
1072 
1073  /* 8 nibble HEX representation */
1074  else
1075  {
1076  /* Only get necessary nibble by ANDing with a mask and
1077  * shifting one nibble (4 bits) per position */
1078  buf[0] = TO_HEX(((value & 0xF0000000) >> 28));
1079  buf[1] = TO_HEX(((value & 0x0F000000) >> 24));
1080  buf[2] = TO_HEX(((value & 0x00F00000) >> 20));
1081  buf[3] = TO_HEX(((value & 0x000F0000) >> 16));
1082 
1083  /* Add spacing if necessary */
1084  if (spacing)
1085  {
1086  buf[4] = ' '; /* Spacing */
1087  buf[5] = TO_HEX(((value & 0x0000F000) >> 12));
1088  buf[6] = TO_HEX(((value & 0x00000F00) >> 8 ));
1089  buf[7] = TO_HEX(((value & 0x000000F0) >> 4 ));
1090  buf[8] = TO_HEX( (value & 0x0000000F) );
1091  buf[9] = '\0'; /* NULL termination character */
1092  }
1093  else
1094  {
1095  buf[4] = TO_HEX(((value & 0x0000F000) >> 12));
1096  buf[5] = TO_HEX(((value & 0x00000F00) >> 8 ));
1097  buf[6] = TO_HEX(((value & 0x000000F0) >> 4 ));
1098  buf[7] = TO_HEX( (value & 0x0000000F) );
1099  buf[8] = '\0'; /* NULL termination character */
1100  }
1101  }
1102 }

1105 /**************************************************************************//**
1106  * @brief
1107  * Convert a `uint32_t` value to a decimal char array (string).
1108  *
1109  * @note
1110  * This is a static method because it's only internally used in this file
1111  * and called by other methods if necessary.
1112  *
1113  * @param[out] buf
1114  * The buffer to put the resulting string in.@n
1115  * **This needs to have a length of 10: `char buf[10];`!**
1116  *
1117  * @param[in] value
1118  * The `uint32_t` value to convert to a string.
1119  *****************************************************************************/
1120 static void uint32_to_charDec (char *buf, uint32_t value)
1121 {
1122  if (value == 0)
1123  {
1124  buf[0] = '0';
1125  buf[1] = '\0'; /* NULL termination character */
1126  }
1127  else
1128  {
1129  /* MAX uint32_t value = FFFFFFFFh = 4294967295d (10 decimal chars) */
1130  char backwardsBuf[10];
1131 
1132  uint32_t calcval = value;
1133  uint8_t length = 0;
1134  uint8_t lengthCounter = 0;
1135 
1136 
1137  /* Loop until the value is zero (separate characters 0-9) and calculate length */
1138  while (calcval)
1139  {
1140  uint32_t rem = calcval % 10;
1141  backwardsBuf[length] = TO_DEC(rem); /* Convert to ASCII character */
1142  length++;
1143 
1144  calcval = calcval - rem;
1145  calcval = calcval / 10;
1146  }
1147 
1148  /* Backwards counter */
1149  lengthCounter = length;
1150 
1151  /* Reverse the characters in the buffer for the final string */
1152  for (uint8_t i = 0; i < length; i++)
1153  {
1154  buf[i] = backwardsBuf[lengthCounter-1];
1155  lengthCounter--;
1156  }
1157 
1158  /* Add NULL termination character */
1159  buf[length] = '\0';
1160  }
1161 }
