37 LoRaStatus_t LoRa_Init(LoRaSettings_t init){
38  PM_Enable(PM_RN2483);
39 
40  RN2483_Init();
41 
42  int retries = 0;
43  while(retries < MAX_JOIN_RETRIES){
44  if(RN2483_Setup(init, loraReceiveBuffer, LORA_BUFFERSIZE) == JOIN_ACCEPTED){
45  break;
46  }
47  retries++;
48 
49 #if DEBUG_DBPRINT == 1 /* DEBUG_DBPRINT */
50  dbwarnInt("Retry join in 5 seconds (LoRa_Init) (", retries, ")\n\r");
51 #endif /* DEBUG_DBPRINT */
52 
53  delay(5000);
54  }
55  if(retries == MAX_JOIN_RETRIES){
56  return (ERROR);
57  }
58  return (JOINED);
59 }
