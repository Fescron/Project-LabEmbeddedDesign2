44 /**************************************************************************//**
45  * @brief
46  * Method to check if the wire is broken.
47  *
48  * @details
49  * This method enables the necessary GPIO clocks, sets the mode of the pins,
50  * checks the connection between them and also disables them at the end.
51  *
52  * @return
53  * @li `true` - The connection is still okay.
54  * @li `false` - The connection is broken!
55  *****************************************************************************/
56 bool checkCable (void)
57 {
58  /* Value to eventually return */
59  bool check = false;
60 
61  /* Enable necessary clocks (just in case) */
62  CMU_ClockEnable(cmuClock_HFPER, true); /* GPIO is a High Frequency Peripheral */
63  CMU_ClockEnable(cmuClock_GPIO, true);
64 
65  /* Change mode of first pin */
66  GPIO_PinModeSet(BREAK1_PORT, BREAK1_PIN, gpioModeInput, 0);
67 
68  /* Change mode of second pin and also set it low with the last argument */
69  GPIO_PinModeSet(BREAK2_PORT, BREAK2_PIN, gpioModePushPull, 0);
70 
71  /* Check the connection */
72  if (!GPIO_PinInGet(BREAK1_PORT,BREAK1_PIN)) check = true;
73 
74  /* Disable the pins */
75  GPIO_PinModeSet(BREAK1_PORT, BREAK1_PIN, gpioModeDisabled, 0);
76  GPIO_PinModeSet(BREAK2_PORT, BREAK2_PIN, gpioModeDisabled, 0);
77 
78  return (check);
79 }
